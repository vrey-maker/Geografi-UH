    let currentQ = 0;
    let score = 0;
    let playerName = "";


    function startQuiz() {
      const nameInput = document.getElementById("username").value.trim();


      if (!nameInput) {
        alert("Masukkan nama dulu ya!");
        return;
      }


      playerName = nameInput;
      document.getElementById("login-screen").classList.add("hidden");
      document.getElementById("quiz-screen").classList.remove("hidden");
      showQuestion();
    }


    function showQuestion() {
      const q = questions[currentQ];


      document.getElementById("progress").innerText = `Soal ${currentQ + 1}/${questions.length}`;
      document.getElementById("score-display").innerText = `Skor: ${score}`;
      document.getElementById("q-text").innerText = q.q;


      const optContainer = document.getElementById("opt-container");
      const essayContainer = document.getElementById("essay-container");


      optContainer.innerHTML = "";


      if (q.type === "essay") {
        optContainer.classList.add("hidden");
        essayContainer.classList.remove("hidden");
      } else {
        optContainer.classList.remove("hidden");
        essayContainer.classList.add("hidden");


        q.options.forEach((opt, index) => {
          const btn = document.createElement("button");
          btn.className = "opt-btn";
          btn.innerText = opt;
          btn.onclick = () => checkAnswer(index);
          optContainer.appendChild(btn);
        });
      }
    }


    function checkAnswer(selected) {
      if (selected === questions[currentQ].correct) {
        score += 4;
      }


      nextQuestion();
    }


    function nextQuestion() {
      if (questions[currentQ].type === "essay") {
        score += 4; // demo saja
      }


      currentQ++;


      if (currentQ < questions.length) {
        showQuestion();
      } else {
        finishQuiz();
      }
    }


    function finishQuiz() {
      document.getElementById("quiz-screen").classList.add("hidden");
      document.getElementById("result-screen").classList.remove("hidden");
      document.getElementById("final-score").innerText = `Skor Kamu: ${score}`;


      saveToLeaderboard();
      displayLeaderboard();
    }


    function saveToLeaderboard() {
      let board = JSON.parse(localStorage.getItem("geoLeaderboard")) || [];


      board.push({ name: playerName, score: score });
      board.sort((a, b) => b.score - a.score);


      localStorage.setItem("geoLeaderboard", JSON.stringify(board.slice(0, 5)));
    }


    function displayLeaderboard() {
      const board = JSON.parse(localStorage.getItem("geoLeaderboard")) || [];
      const body = document.getElementById("leaderboard-body");


      body.innerHTML = board
        .map(
          (entry, i) => `
            <tr>
              <td class="rank">#${i + 1}</td>
              <td>${entry.name}</td>
              <td>${entry.score}</td>
            </tr>
          `
        )
        .join("");
    }
